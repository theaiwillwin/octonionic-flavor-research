def import_path(filename: str, module_name: str):
    path = (ROOT / filename).resolve()
    spec = importlib.util.spec_from_file_location(module_name, path)
    if spec is None or spec.loader is None:
        raise AssertionError(f"cannot create import spec: {path}")
    module = importlib.util.module_from_spec(spec)
    sys.modules[module_name] = module
    spec.loader.exec_module(module)
    actual = Path(module.__file__).resolve()
    if actual != path:
        raise AssertionError(f"wrong module path: {actual} != {path}")
    return module

# Verify the corrected theorem and its necessary Lagrangian hypothesis.
reach_module = modules["next_hard_gate/test_pmns_polar_link_reachability_v2.py"]
phi_reach = reach_module.build_phi()
h_reach = np.zeros(7)
h_reach[0] = 1.0
j_reach = np.einsum("ijk,k->ij", phi_reach, h_reach)
p_reach, q_reach = reach_module.construct_complex_basis(j_reach, h_reach)
if np.linalg.norm(p_reach.T @ j_reach @ p_reach) > 1.0e-12:
    raise AssertionError("corrected construction is not Lagrangian")
a_reach = np.eye(7) - np.outer(h_reach, h_reach) + 1j * j_reach
rng_reach = np.random.default_rng(20260714)
worst_reach = 0.0
for _ in range(100):
    target = reach_module.haar_unitary(rng_reach)
    lnu = reach_module.frame_for_unitary(target, p_reach, q_reach)
    residual = np.linalg.norm(p_reach.T @ a_reach @ lnu - target)
    worst_reach = max(worst_reach, float(residual))
if worst_reach > 1.0e-12:
    raise AssertionError(f"corrected U(3) construction residual: {worst_reach}")

# Exhibit why arbitrary orthonormal P is insufficient.
p_bad = np.column_stack((p_reach[:, 0], q_reach[:, 0], p_reach[:, 1]))
if np.linalg.norm(p_bad.T @ p_bad - np.eye(3)) > 1.0e-12:
    raise AssertionError("counterexample P is not orthonormal")
bad_lagrangian = float(np.linalg.norm(p_bad.T @ j_reach @ p_bad))
if bad_lagrangian < 1.0:
    raise AssertionError("counterexample did not violate the Lagrangian condition")
q_bad = -j_reach @ p_bad
target_bad = reach_module.haar_unitary(rng_reach)
lnu_bad = reach_module.frame_for_unitary(target_bad, p_bad, q_bad)
bad_residual = float(np.linalg.norm(p_bad.T @ a_reach @ lnu_bad - target_bad))
if bad_residual < 1.0e-3:
    raise AssertionError("arbitrary orthonormal counterexample unexpectedly satisfied K=U")
